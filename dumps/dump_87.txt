<script>
var catalogMenuPath = [0], catalogMenuHeight,
	listId = 0;


/*------------------------------------------------------------------------------------------*/

$('#catalog_menu').on('click', 'a.next', function() {
	var _this = $(this),
		id = _this.data('id'),
		_id = '#menucat'+id.toString(10).replace('.','\\.'),
		m = $(_id),
		_show = function() {
			catalogMenuHeight = m.height()-2/*border*/;
			$('#catalog_menu').height(catalogMenuHeight);
			m.css({left:0});
			if (m.offset().top < $(window).scrollTop()) {
				$('html,body').animate({scrollTop:m.offset().top}, 500);
			}
		};

	catalogMenuPath.push(id);

	if (!m.length) {
		_this.addClass('wait');
		$.get('/ajax/catalog_menu.php', {id:id, list:listId}, function(res) {
			$('#catalog_menu').append(res);
			m = $(_id);
			_show();
			_this.removeClass('wait');
		}, 'html');
	}
	else {
		_show();
	}
});

/*------------------------------------------------------------------------------------------*/

$('#catalog_menu').on('click', 'a.prev', function () {
	var _this = $(this),
		m = $(this).closest('.menu'),
		id = catalogMenuPath[catalogMenuPath.length - 2],
		_id = '#menucat'+id.toString(10).replace('.','\\.'),
		_show = function() {
			catalogMenuHeight = $(_id).height()-2/*border*/;
			$('#catalog_menu').height(catalogMenuHeight);
			m.css({left:'100%'});
		};

	catalogMenuPath.pop();

	if (!$(_id).length) {
		_this.addClass('wait');
		$.get('/ajax/catalog_menu.php', {id:id, list:listId}, function(res) {
			$('#catalog_menu').prepend(res);
			$(_id).css({left:0});
			_show();
			_this.removeClass('wait');
		}, 'html');
	}
	else {
		_show();
	}
});

/*------------------------------------------------------------------------------------------*/

$('#catalog_menu_mob_btn').click(function() {
	if ($(this).hasClass('active')) {
		$('#catalog_menu_xs').css({height: catalogMenuHeight+15/*margin*/});
		setTimeout(function() { // wait animation
			$('#catalog_menu_xs').css({height: 'auto'});
		}, 500);
	}
	else {
		$('#catalog_menu_xs').height(catalogMenuHeight);
		$('#catalog_menu_xs').height(0);
	}

});
</script>
